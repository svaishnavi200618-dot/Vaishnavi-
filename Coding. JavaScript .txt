// Greeting Message (Dynamic Based on Time)
window.onload = function() {
  let greeting;
  const hour = new Date().getHours();

  if (hour < 12) {
    greeting = "Good Morning 🌅";
  } else if (hour < 18) {
    greeting = "Good Afternoon ☀️";
  } else {
    greeting = "Good Evening 🌙";
  }

  const home = document.querySelector("#home h2");
  home.innerText = `${greeting}, I am Vaishnavi 👋`;
};
